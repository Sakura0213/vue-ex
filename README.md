``` shell

//安装 nodejs
ui:https://nodejs.org/en
node -v
npm -v

//安装yarn，-g表示全局安装
npm install yarn -g
//安装 pnpm
npm install pnpm -g
//安装vite，构建工具
npm install vite -g

//创建 vue 项目
npm init vue@latest
cd 创建的工程
npm install
npm run dev

//安装依赖
//axios（网络请求库）
yarn add axios
//mockjs（模拟接口）
npm install mockjs
//element-plus（ui库）
yarn add element-plus

//若在package.json中已经引用好依赖后
npm install ？？
```
